```sh
# or use pnpm
pnpm add unplugin-operator-overload -D
```

### Defining Methods

Like Python, use magic methods such as `__add__`, `__sub__` to define operator overloading:

```ts title="example.ts"
class Vector {
  constructor(
    public x: number,
    public y: number,
  ) {}
  __add__(other: Vector) {
    return new Vector(this.x + other.x, this.y + other.y);
  }
  __sub__(other: Vector) {
    return new Vector(this.x - other.x, this.y - other.y);
  }
  __mul__(scalar: number) {
    return new Vector(this.x * scalar, this.y * scalar);
  }
  __div__(scalar: number) {
    return new Vector(this.x / scalar, this.y / scalar);
  }
}
```
